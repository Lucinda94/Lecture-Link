/**
 * Displays the messages of a specific user in the chat container
 * @param {integer} participant_1_id - The id of the current user
 * @param {integer} participant_2_id - The id of the other user in the conversation
 */
function displayMessages(participant_1_id, participant_2_id) {
    const messages = getMessages(participant_1_id, participant_2_id);
    for (message of messages) {
        // check if message is inbound or outbound
        if (message.user_id === participant_1_id) {
            appendMessageToChat(message, true);
        } else {
            appendMessageToChat(message, false);
        }
    }
}
